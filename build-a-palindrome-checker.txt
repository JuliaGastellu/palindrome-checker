    <script>
        const checkPalindrome = () => {
            const input = document.getElementById('text-input').value;
            if (!input.trim()) {
                alert('Please input a value');
                return;
            }
            
            // Clean the input: lowercase and remove non-alphanumeric
            const cleaned = input
                .toLowerCase()
                .replace(/[^a-z0-9]/g, '');
            
            const isPalindrome = cleaned === [...cleaned].reverse().join('');
            const resultText = `${input} ${isPalindrome ? 'is a palindrome' : 'is not a palindrome'}`;
            
            document.getElementById('result').textContent = resultText;
        };

        document.getElementById('check-btn').addEventListener('click', checkPalindrome);
